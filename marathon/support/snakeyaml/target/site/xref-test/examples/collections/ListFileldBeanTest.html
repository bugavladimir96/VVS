28  /***
29   * Test public field ListBean->List<Developer> developers <br/>
30   * Developer class must be properly recognised
31   */
32  public class ListFileldBeanTest extends TestCase {
33      public void testDumpList() {
34          ListFieldBean bean = new ListFieldBean();
35          List<String> list = new ArrayList<String>();
36          list.add("aaa");
37          list.add("bbb");
38          bean.setChildren(list);
39          List<Developer> developers = new ArrayList<Developer>();
40          developers.add(new Developer("Fred", "creator"));
41          developers.add(new Developer("John", "committer"));
42          bean.developers = developers;
43          bean.setName("Bean123");
44          JavaBeanDumper dumper = new JavaBeanDumper(false);
45          String output = dumper.dump(bean);
46          // System.out.println(output);
47          String etalon = Util.getLocalResource("examples/list-bean-1.yaml");
48          assertEquals(etalon, output);
49      }
50  
51      public void testLoadList() {
52          String output = Util.getLocalResource("examples/list-bean-1.yaml");
53          // System.out.println(output);
54          JavaBeanLoader<ListFieldBean> beanLoader = new JavaBeanLoader<ListFieldBean>(
55                  ListFieldBean.class);
56          ListFieldBean parsed = beanLoader.load(output);
57          assertNotNull(parsed);
58          List<String> list2 = parsed.getChildren();
59          assertEquals(2, list2.size());
60          assertEquals("aaa", list2.get(0));
61          assertEquals("bbb", list2.get(1));
62          List<Developer> developers = parsed.developers;
63          assertEquals(2, developers.size());
64          assertEquals("Developer must be recognised.", Developer.class, developers.get(0).getClass());
65          Developer fred = developers.get(0);
66          assertEquals("Fred", fred.getName());
67          assertEquals("creator", fred.getRole());
68      }
