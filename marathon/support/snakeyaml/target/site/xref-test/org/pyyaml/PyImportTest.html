89      protected List<Event> canonicalParse(InputStream input2) throws IOException {
90          StreamReader reader = new StreamReader(new UnicodeReader(input2));
91          StringBuilder buffer = new StringBuilder();
92          while (reader.peek() != '\0') {
93              buffer.append(reader.peek());
94              reader.forward();
95          }
96          CanonicalParser parser = new CanonicalParser(buffer.toString());
97          List<Event> result = new ArrayList<Event>();
98          while (parser.peekEvent() != null) {
99              result.add(parser.getEvent());
100         }

114     private class PyFilenameFilter implements FilenameFilter {
115         private String extension;
116         private boolean onlyIfCanonicalPresent;
117 
118         public PyFilenameFilter(String extension, boolean onlyIfCanonicalPresent) {
119             this.extension = extension;
120             this.onlyIfCanonicalPresent = onlyIfCanonicalPresent;
121         }
122 
123         public boolean accept(File dir, String name) {
124             int position = name.lastIndexOf('.');
125             String canonicalFileName = name.substring(0, position) + ".canonical";
126             File canonicalFile = new File(dir, canonicalFileName);
127             if (onlyIfCanonicalPresent && !canonicalFile.exists()) {
128                 return false;
129             } else {
130                 return name.endsWith(extension);
131             }
132         }
