99      public void testNoRootTag() throws Exception {
100         JavaBeanWithNullValues javaBeanWithNullValues = new JavaBeanWithNullValues();
101         DumperOptions options = new DumperOptions();
102         options.setDefaultScalarStyle(DumperOptions.ScalarStyle.DOUBLE_QUOTED);
103         options.setExplicitStart(true);
104         options.setExplicitEnd(true);
105         options.setExplicitRoot(Tag.MAP);
106         Yaml yaml = new Yaml(new CustomRepresenter(), options);
107         javaBeanWithNullValues.setBoolean1(null);
108         javaBeanWithNullValues.setDate(new Date(System.currentTimeMillis()));
109         javaBeanWithNullValues.setDouble1(1d);
110         javaBeanWithNullValues.setFloat1(1f);
111         javaBeanWithNullValues.setInteger(1);
112         javaBeanWithNullValues.setLong1(1l);
113         javaBeanWithNullValues.setSqlDate(new java.sql.Date(System.currentTimeMillis()));
114         javaBeanWithNullValues.setString(null); // ok
115         javaBeanWithNullValues.setTimestamp(new Timestamp(System.currentTimeMillis()));
116 
117         String dumpStr = yaml.dump(javaBeanWithNullValues);
118         // System.out.println(dumpStr);
119         assertFalse("No explicit root tag must be used.", dumpStr
120                 .contains("JavaBeanWithNullValues"));
121         yaml = new Yaml(new CustomRepresenter(), options);
122         JavaBeanWithNullValues parsed = loader.load(dumpStr);
123         assertNull(" expect null, got " + parsed.getBoolean1(), parsed.getBoolean1());
124         assertNull(" expect null, got " + parsed.getString(), parsed.getString());
125         assertEquals(1d, parsed.getDouble1());
126         assertEquals(1f, parsed.getFloat1());
127         assertEquals(new Integer(1), parsed.getInteger());
128         assertEquals(new Long(1l), parsed.getLong1());
129     }

131     private String dumpJavaBeanWithNullValues(boolean nullValues) {
132         JavaBeanWithNullValues javaBeanWithNullValues = new JavaBeanWithNullValues();
133         DumperOptions options = new DumperOptions();
134         options.setDefaultScalarStyle(DumperOptions.ScalarStyle.DOUBLE_QUOTED);
135         options.setExplicitStart(true);
136         options.setExplicitEnd(true);
137         Yaml yaml = new Yaml(new CustomRepresenter(), options);
138         if (nullValues) {
139             return yaml.dump(javaBeanWithNullValues);
140         }
141         javaBeanWithNullValues.setBoolean1(false);
142         javaBeanWithNullValues.setDate(new Date(System.currentTimeMillis()));
143         javaBeanWithNullValues.setDouble1(1d);
144         javaBeanWithNullValues.setFloat1(1f);
145         javaBeanWithNullValues.setInteger(1);
146         javaBeanWithNullValues.setLong1(1l);
147         javaBeanWithNullValues.setSqlDate(new java.sql.Date(System.currentTimeMillis()));
148         javaBeanWithNullValues.setString(""); // ok
149         javaBeanWithNullValues.setTimestamp(new Timestamp(System.currentTimeMillis()));
150         return yaml.dump(javaBeanWithNullValues);
151     }

153     public class CustomRepresenter extends Representer {
154         public CustomRepresenter() {
155             this.representers.put(Float.class, new RepresentFloat());
156             this.representers.put(Long.class, new RepresentLong());
157             this.representers.put(java.sql.Date.class, new RepresentDate());
158             this.representers.put(java.sql.Timestamp.class, new RepresentTime());
159         }
160 
161         private class RepresentFloat implements Represent {
162             public Node representData(Object data) {
163                 return representScalar(new Tag(Tag.PREFIX + "java.lang.Float"), ((Float) data)
164                         .toString());
165             }
166         }
167 
168         private class RepresentLong implements Represent {
169             public Node representData(Object data) {
170                 return representScalar(new Tag(Tag.PREFIX + "java.lang.Long"), ((Long) data)
171                         .toString());
172             }
173         }
174 
175         private class RepresentDate implements Represent {
176             public Node representData(Object data) {
177                 return representScalar(new Tag(Tag.PREFIX + "java.sql.Date"),
178                         ((java.sql.Date) data).toString());
179             }
180         }
181 
182         private class RepresentTime implements Represent {
183             public Node representData(Object data) {
184                 return representScalar(new Tag(Tag.PREFIX + "java.sql.Timestamp"),
185                         ((java.sql.Timestamp) data).toString());
186             }
187         }
