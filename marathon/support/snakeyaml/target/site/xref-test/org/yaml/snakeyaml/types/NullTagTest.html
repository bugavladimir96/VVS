37      public void testNull() throws IOException {
38          assertNull("Got: '" + load("---\n") + "'", load("---\n"));
39          assertNull(load("---\n..."));
40          assertNull(load("---\n...\n"));
41          assertNull(load("\n"));
42          assertNull(load(""));
43          assertNull(load(" "));
44          assertNull(load("~"));
45          assertNull(load("---\n~"));
46          assertNull(load("null"));
47          assertNull(load("Null"));
48          assertNull(load("NULL"));
49          assertNull(getMapValue("empty:\n", "empty"));
50          assertNull(getMapValue("canonical: ~", "canonical"));
51          assertNull(getMapValue("english: null", "english"));
52          assertNull(getMapValue("english: Null", "english"));
53          assertNull(getMapValue("english: NULL", "english"));
54          assertEquals("null key", getMapValue("~: null key\n", null));
55      }

57      @SuppressWarnings("unchecked")
58      public void testSequenceNull() throws IOException {
59          String input = "---\n# This sequence has five\n# entries, two have values.\nsparse:\n  - ~\n  - 2nd entry\n  -\n  - 4th entry\n  - Null\n";
60          List<String> parsed = (List<String>) getMapValue(input, "sparse");
61          assertEquals(5, parsed.size());
62          assertNull(parsed.get(0));
63          assertEquals("2nd entry", parsed.get(1));
64          assertNull("Got: '" + parsed.get(2) + "'", parsed.get(2));
65          assertEquals("4th entry", parsed.get(3));
66          assertNull(parsed.get(4));
67      }

69      public void testNullInMap() throws IOException {
70          String input = "key1: null\n~: value1";
71          Map<String, Object> parsed = getMap(input);
72          assertEquals(2, parsed.size());
73          assertTrue(parsed.containsKey(null));
74          Object value1 = parsed.get(null);
75          assertEquals("value1", value1);
76          //
77          assertNull(parsed.get("key1"));
78          //
79          assertFalse(getMap("key2: value2").containsKey(null));
80      }

127     private class NullRepresenter extends Representer {
128         public NullRepresenter() {
129             super();
130             // null representer is exceptional and it is stored as an instance

132             this.nullRepresenter = new RepresentNull();

135         private class RepresentNull implements Represent {
136             public Node representData(Object data) {
