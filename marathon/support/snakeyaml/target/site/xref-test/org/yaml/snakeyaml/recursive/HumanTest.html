105         Human mother = new Human();

112         //
113         Human son = new Human();
114         son.setName("Son");

116         son.setBirthPlace("Munich");
117         son.setBankAccountOwner(father);
118         son.setFather(father);
119         son.setMother(mother);
120         //
121         Human daughter = new Human();
122         daughter.setName("Daughter");
123         daughter.setBirthday(new Date(420000000000L));
124         daughter.setBirthPlace("New York");
125         daughter.setBankAccountOwner(father);
126         daughter.setFather(father);
127         daughter.setMother(mother);
128         //
129         Set<Human> children = new LinkedHashSet<Human>(2);
130         children.add(son);
131         children.add(daughter);
132         father.setChildren(children);
133         mother.setChildren(children);
134         //
135         JavaBeanDumper beanDumper = new JavaBeanDumper();
136         String output = beanDumper.dump(son);
137         // System.out.println(output);
138         String etalon = Util.getLocalResource("recursive/with-children.yaml");
139         assertEquals(etalon, output);
140         TypeDescription humanDescription = new TypeDescription(Human.class);
141         humanDescription.putMapPropertyType("children", Human.class, Object.class);
142         JavaBeanLoader<Human> beanLoader = new JavaBeanLoader<Human>(humanDescription);

144         Human son2 = beanLoader.load(output);
145         assertNotNull(son2);
146         assertEquals("Son", son.getName());
147 
148         Human father2 = son2.getFather();
149         assertEquals("Father", father2.getName());
150         assertEquals("Mother", son2.getMother().getName());
151         assertSame(father2, father2.getBankAccountOwner());
152         assertSame(father2.getPartner(), son2.getMother());
153         assertSame(father2, son2.getMother().getPartner());
154 
155         Set<Human> children2 = father2.getChildren();
156         assertEquals(2, children2.size());
157         assertSame(father2.getPartner().getChildren(), children2);
158 
159         for (Object child : children2) {
160             // check if type descriptor was correct
161             assertSame(Human.class, child.getClass());
162         }
163 
164         // check if hashCode is correct
165         validateSet(children2);
166     }
