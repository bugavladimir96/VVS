27  public class SetTagTest extends AbstractTest {

29      @SuppressWarnings("unchecked")
30      public void testSet() {
31          YamlDocument document = new YamlDocument("types/set.yaml");
32          Map<String, Set<String>> map = (Map<String, Set<String>>) document.getNativeData();
33          assertEquals(2, map.size());
34          Set<String> set1 = (Set<String>) map.get("baseball players");
35          assertEquals(3, set1.size());
36          assertTrue(set1.contains("Mark McGwire"));
37          assertTrue(set1.contains("Sammy Sosa"));
38          assertTrue(set1.contains("Ken Griffey"));
39          //
40          Set<String> set2 = (Set<String>) map.get("baseball teams");
41          assertEquals(3, set2.size());
42          assertTrue(set2.contains("Boston Red Sox"));
43          assertTrue(set2.contains("Detroit Tigers"));
44          assertTrue(set2.contains("New York Yankees"));
45      }
