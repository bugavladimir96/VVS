36      private String content;

41  
42      public boolean has(char ch) {
43          return content.indexOf(ch) != -1;
44      }
45  
46      public boolean hasNo(char ch) {
47          return !has(ch);
48      }
49  
50      public boolean has(char ch, String additional) {
51          return additional.indexOf(ch) != -1 || content.indexOf(ch) != -1;
52      }
53  
54      public boolean hasNo(char ch, String additional) {
55          return !has(ch, additional);
