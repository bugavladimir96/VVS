54  /***
55   * <pre>
56   * # The following YAML grammar is LL(1) and is parsed by a recursive descent
57   * parser.
58   * stream            ::= STREAM-START implicit_document? explicit_document* STREAM-END
59   * implicit_document ::= block_node DOCUMENT-END*
60   * explicit_document ::= DIRECTIVE* DOCUMENT-START block_node? DOCUMENT-END*
61   * block_node_or_indentless_sequence ::=
62   *                       ALIAS
63   *                       | properties (block_content | indentless_block_sequence)?
64   *                       | block_content
65   *                       | indentless_block_sequence
66   * block_node        ::= ALIAS
67   *                       | properties block_content?
68   *                       | block_content
69   * flow_node         ::= ALIAS
70   *                       | properties flow_content?
71   *                       | flow_content
72   * properties        ::= TAG ANCHOR? | ANCHOR TAG?
73   * block_content     ::= block_collection | flow_collection | SCALAR
74   * flow_content      ::= flow_collection | SCALAR
75   * block_collection  ::= block_sequence | block_mapping
76   * flow_collection   ::= flow_sequence | flow_mapping
77   * block_sequence    ::= BLOCK-SEQUENCE-START (BLOCK-ENTRY block_node?)* BLOCK-END
78   * indentless_sequence   ::= (BLOCK-ENTRY block_node?)+
79   * block_mapping     ::= BLOCK-MAPPING_START
80   *                       ((KEY block_node_or_indentless_sequence?)?
81   *                       (VALUE block_node_or_indentless_sequence?)?)*
82   *                       BLOCK-END
83   * flow_sequence     ::= FLOW-SEQUENCE-START
84   *                       (flow_sequence_entry FLOW-ENTRY)*
85   *                       flow_sequence_entry?
86   *                       FLOW-SEQUENCE-END
87   * flow_sequence_entry   ::= flow_node | KEY flow_node? (VALUE flow_node?)?
88   * flow_mapping      ::= FLOW-MAPPING-START
89   *                       (flow_mapping_entry FLOW-ENTRY)*
90   *                       flow_mapping_entry?
91   *                       FLOW-MAPPING-END
92   * flow_mapping_entry    ::= flow_node | KEY flow_node? (VALUE flow_node?)?
93   * FIRST sets:
94   * stream: { STREAM-START }
95   * explicit_document: { DIRECTIVE DOCUMENT-START }
96   * implicit_document: FIRST(block_node)
97   * block_node: { ALIAS TAG ANCHOR SCALAR BLOCK-SEQUENCE-START BLOCK-MAPPING-START FLOW-SEQUENCE-START FLOW-MAPPING-START }
98   * flow_node: { ALIAS ANCHOR TAG SCALAR FLOW-SEQUENCE-START FLOW-MAPPING-START }
99   * block_content: { BLOCK-SEQUENCE-START BLOCK-MAPPING-START FLOW-SEQUENCE-START FLOW-MAPPING-START SCALAR }
100  * flow_content: { FLOW-SEQUENCE-START FLOW-MAPPING-START SCALAR }
101  * block_collection: { BLOCK-SEQUENCE-START BLOCK-MAPPING-START }
102  * flow_collection: { FLOW-SEQUENCE-START FLOW-MAPPING-START }
103  * block_sequence: { BLOCK-SEQUENCE-START }
104  * block_mapping: { BLOCK-MAPPING-START }
105  * block_node_or_indentless_sequence: { ALIAS ANCHOR TAG SCALAR BLOCK-SEQUENCE-START BLOCK-MAPPING-START FLOW-SEQUENCE-START FLOW-MAPPING-START BLOCK-ENTRY }
106  * indentless_sequence: { ENTRY }
107  * flow_collection: { FLOW-SEQUENCE-START FLOW-MAPPING-START }
108  * flow_sequence: { FLOW-SEQUENCE-START }
109  * flow_mapping: { FLOW-MAPPING-START }
110  * flow_sequence_entry: { ALIAS ANCHOR TAG SCALAR FLOW-SEQUENCE-START FLOW-MAPPING-START KEY }
111  * flow_mapping_entry: { ALIAS ANCHOR TAG SCALAR FLOW-SEQUENCE-START FLOW-MAPPING-START KEY }
112  * </pre>
113  * 
114  * Since writing a recursive-descendant parser is a straightforward task, we do
115  * not give many comments here.
116  * 
117  * @see <a href="http://pyyaml.org/wiki/PyYAML">PyYAML</a> for more information
118  */
119 public final class ParserImpl implements Parser {
120     private static final Map<String, String> DEFAULT_TAGS = new HashMap<String, String>();
121     static {
122         DEFAULT_TAGS.put("!", "!");
123         DEFAULT_TAGS.put("!!", Tag.PREFIX);
124     }

126     private final Scanner scanner;
127     private Event currentEvent;
128     private List<Integer> yamlVersion;

134     public ParserImpl(StreamReader reader) {
135         this.scanner = new ScannerImpl(reader);
136         currentEvent = null;
137         yamlVersion = null;
138         tagHandles = new HashMap<String, String>();
139         states = new ArrayStack<Production>(100);
140         marks = new ArrayStack<Mark>(10);
141         state = new ParseStreamStart();
142     }

144     /***
145      * Check the type of the next event.
146      */
147     public boolean checkEvent(Event.ID choices) {
148         peekEvent();
149         if (currentEvent != null) {
150             if (currentEvent.is(choices)) {
151                 return true;
152             }
153         }
154         return false;
155     }
156 
157     /*
158      * Get the next event.
159      */
160     public Event peekEvent() {
161         if (currentEvent == null) {
162             if (state != null) {
163                 currentEvent = state.produce();
164             }
165         }
166         return currentEvent;
167     }
168 
169     /*
170      * Get the next event and proceed further.
171      */
172     public Event getEvent() {
173         peekEvent();
174         Event value = currentEvent;
175         currentEvent = null;
176         return value;
177     }

179     /***
180      * <pre>
181      * stream    ::= STREAM-START implicit_document? explicit_document* STREAM-END
182      * implicit_document ::= block_node DOCUMENT-END*
183      * explicit_document ::= DIRECTIVE* DOCUMENT-START block_node? DOCUMENT-END*
184      * </pre>
185      */
186     private class ParseStreamStart implements Production {
187         public Event produce() {
188             // Parse the stream start.
189             StreamStartToken token = (StreamStartToken) scanner.getToken();
190             Event event = new StreamStartEvent(token.getStartMark(), token.getEndMark());
191             // Prepare the next state.
192             state = new ParseImplicitDocumentStart();
193             return event;
194         }
195     }

197     private class ParseImplicitDocumentStart implements Production {
198         public Event produce() {
199             // Parse an implicit document.
200             if (!scanner.checkToken(Token.ID.Directive, Token.ID.DocumentStart, Token.ID.StreamEnd)) {
201                 tagHandles = DEFAULT_TAGS;
202                 Token token = scanner.peekToken();
203                 Mark startMark = token.getStartMark();
204                 Mark endMark = startMark;
205                 Event event = new DocumentStartEvent(startMark, endMark, false, null, null);
206                 // Prepare the next state.
207                 states.push(new ParseDocumentEnd());
208                 state = new ParseBlockNode();
209                 return event;
210             } else {
211                 Production p = new ParseDocumentStart();
212                 return p.produce();
213             }
214         }
215     }

217     private class ParseDocumentStart implements Production {
218         @SuppressWarnings("unchecked")
219         public Event produce() {
220             // Parse any extra document end indicators.
221             while (scanner.checkToken(Token.ID.DocumentEnd)) {
222                 scanner.getToken();
223             }
224             // Parse an explicit document.
225             Event event;
226             if (!scanner.checkToken(Token.ID.StreamEnd)) {
227                 Token token = scanner.peekToken();
228                 Mark startMark = token.getStartMark();
229                 List<Object> version_tags = processDirectives();
230                 List<Object> version = (List<Object>) version_tags.get(0);
231                 Map<String, String> tags = (Map<String, String>) version_tags.get(1);
232                 if (!scanner.checkToken(Token.ID.DocumentStart)) {
233                     throw new ParserException(null, null, "expected '<document start>', but found "
234                             + scanner.peekToken().getTokenId(), scanner.peekToken().getStartMark());
235                 }
236                 token = scanner.getToken();
237                 Mark endMark = token.getEndMark();
238                 Integer[] versionInteger;
239                 if (version != null) {
240                     versionInteger = new Integer[2];
241                     versionInteger = version.toArray(versionInteger);
242                 } else {
243                     versionInteger = null;
244                 }
245                 event = new DocumentStartEvent(startMark, endMark, true, versionInteger, tags);
246                 states.push(new ParseDocumentEnd());
247                 state = new ParseDocumentContent();
248             } else {
249                 // Parse the end of the stream.
250                 StreamEndToken token = (StreamEndToken) scanner.getToken();
251                 event = new StreamEndEvent(token.getStartMark(), token.getEndMark());
252                 if (!states.isEmpty()) {
253                     throw new YAMLException("Unexpected end of stream. States left: " + states);
254                 }
255                 if (!marks.isEmpty()) {
256                     throw new YAMLException("Unexpected end of stream. Marks left: " + marks);
257                 }
258                 state = null;
259             }
260             return event;
261         }
262     }

264     private class ParseDocumentEnd implements Production {
265         public Event produce() {
266             // Parse the document end.
267             Token token = scanner.peekToken();
268             Mark startMark = token.getStartMark();
269             Mark endMark = startMark;
270             boolean explicit = false;
271             if (scanner.checkToken(Token.ID.DocumentEnd)) {
272                 token = scanner.getToken();
273                 endMark = token.getEndMark();
274                 explicit = true;
275             }
276             Event event = new DocumentEndEvent(startMark, endMark, explicit);
277             // Prepare the next state.
278             state = new ParseDocumentStart();
279             return event;
280         }
281     }

283     private class ParseDocumentContent implements Production {
284         public Event produce() {
285             Event event;
286             if (scanner.checkToken(Token.ID.Directive, Token.ID.DocumentStart,
287                     Token.ID.DocumentEnd, Token.ID.StreamEnd)) {
288                 event = processEmptyScalar(scanner.peekToken().getStartMark());
289                 state = states.pop();
290                 return event;
291             } else {
292                 Production p = new ParseBlockNode();
293                 return p.produce();
294             }
295         }
296     }

298     @SuppressWarnings("unchecked")
299     private List<Object> processDirectives() {
300         yamlVersion = null;
301         tagHandles = new HashMap<String, String>();
302         while (scanner.checkToken(Token.ID.Directive)) {
303             DirectiveToken token = (DirectiveToken) scanner.getToken();
304             if (token.getName().equals("YAML")) {
305                 if (yamlVersion != null) {
306                     throw new ParserException(null, null, "found duplicate YAML directive", token
307                             .getStartMark());
308                 }
309                 List<Integer> value = (List<Integer>) token.getValue();
310                 Integer major = value.get(0);
311                 if (major != 1) {
312                     throw new ParserException(null, null,
313                             "found incompatible YAML document (version 1.* is required)", token
314                                     .getStartMark());
315                 }
316                 yamlVersion = (List<Integer>) token.getValue();
317             } else if (token.getName().equals("TAG")) {
318                 List<String> value = (List<String>) token.getValue();
319                 String handle = value.get(0);
320                 String prefix = value.get(1);
321                 if (tagHandles.containsKey(handle)) {
322                     throw new ParserException(null, null, "duplicate tag handle " + handle, token
323                             .getStartMark());
324                 }
325                 tagHandles.put(handle, prefix);
326             }
327         }
328         List<Object> value = new ArrayList<Object>(2);
329         value.add(yamlVersion);
330         if (!tagHandles.isEmpty()) {
331             value.add(new HashMap<String, String>(tagHandles));
332         } else {
333             value.add(new HashMap<String, String>());
334         }
335         for (String key : DEFAULT_TAGS.keySet()) {
336             if (!tagHandles.containsKey(key)) {
337                 tagHandles.put(key, DEFAULT_TAGS.get(key));
338             }
339         }
340         return value;
341     }

377     private Event parseNode(boolean block, boolean indentlessSequence) {
378         Event event;
379         Mark startMark = null;
380         Mark endMark = null;
381         Mark tagMark = null;
382         if (scanner.checkToken(Token.ID.Alias)) {
383             AliasToken token = (AliasToken) scanner.getToken();
384             event = new AliasEvent(token.getValue(), token.getStartMark(), token.getEndMark());
385             state = states.pop();
386         } else {
387             String anchor = null;
388             TagTuple tagTokenTag = null;
389             if (scanner.checkToken(Token.ID.Anchor)) {
390                 AnchorToken token = (AnchorToken) scanner.getToken();
391                 startMark = token.getStartMark();
392                 endMark = token.getEndMark();
393                 anchor = token.getValue();
394                 if (scanner.checkToken(Token.ID.Tag)) {
395                     TagToken tagToken = (TagToken) scanner.getToken();
396                     tagMark = tagToken.getStartMark();
397                     endMark = tagToken.getEndMark();
398                     tagTokenTag = tagToken.getValue();
399                 }
400             } else if (scanner.checkToken(Token.ID.Tag)) {
401                 TagToken tagToken = (TagToken) scanner.getToken();
402                 startMark = tagToken.getStartMark();
403                 tagMark = startMark;
404                 endMark = tagToken.getEndMark();
405                 tagTokenTag = tagToken.getValue();
406                 if (scanner.checkToken(Token.ID.Anchor)) {
407                     AnchorToken token = (AnchorToken) scanner.getToken();
408                     endMark = token.getEndMark();
409                     anchor = token.getValue();
410                 }
411             }
412             String tag = null;
413             if (tagTokenTag != null) {
414                 String handle = tagTokenTag.getHandle();
415                 String suffix = tagTokenTag.getSuffix();
416                 if (handle != null) {
417                     if (!tagHandles.containsKey(handle)) {
418                         throw new ParserException("while parsing a node", startMark,
419                                 "found undefined tag handle " + handle, tagMark);
420                     }
421                     tag = tagHandles.get(handle) + suffix;
422                 } else {
423                     tag = suffix;
424                 }
425             }
426             if (startMark == null) {
427                 startMark = scanner.peekToken().getStartMark();
428                 endMark = startMark;
429             }
430             event = null;
431             boolean implicit = (tag == null || tag.equals("!"));
432             if (indentlessSequence && scanner.checkToken(Token.ID.BlockEntry)) {
433                 endMark = scanner.peekToken().getEndMark();
434                 event = new SequenceStartEvent(anchor, tag, implicit, startMark, endMark,
435                         Boolean.FALSE);
436                 state = new ParseIndentlessSequenceEntry();
437             } else {
438                 if (scanner.checkToken(Token.ID.Scalar)) {
439                     ScalarToken token = (ScalarToken) scanner.getToken();
440                     endMark = token.getEndMark();
441                     ImplicitTuple implicitValues;
442                     if ((token.getPlain() && tag == null) || "!".equals(tag)) {
443                         implicitValues = new ImplicitTuple(true, false);
444                     } else if (tag == null) {
445                         implicitValues = new ImplicitTuple(false, true);
446                     } else {
447                         implicitValues = new ImplicitTuple(false, false);
448                     }
449                     event = new ScalarEvent(anchor, tag, implicitValues, token.getValue(),
450                             startMark, endMark, token.getStyle());
451                     state = states.pop();
452                 } else if (scanner.checkToken(Token.ID.FlowSequenceStart)) {
453                     endMark = scanner.peekToken().getEndMark();
454                     event = new SequenceStartEvent(anchor, tag, implicit, startMark, endMark,
455                             Boolean.TRUE);
456                     state = new ParseFlowSequenceFirstEntry();
457                 } else if (scanner.checkToken(Token.ID.FlowMappingStart)) {
458                     endMark = scanner.peekToken().getEndMark();
459                     event = new MappingStartEvent(anchor, tag, implicit, startMark, endMark,
460                             Boolean.TRUE);
461                     state = new ParseFlowMappingFirstKey();
462                 } else if (block && scanner.checkToken(Token.ID.BlockSequenceStart)) {
463                     endMark = scanner.peekToken().getStartMark();
464                     event = new SequenceStartEvent(anchor, tag, implicit, startMark, endMark,
465                             Boolean.FALSE);
466                     state = new ParseBlockSequenceFirstEntry();
467                 } else if (block && scanner.checkToken(Token.ID.BlockMappingStart)) {
468                     endMark = scanner.peekToken().getStartMark();
469                     event = new MappingStartEvent(anchor, tag, implicit, startMark, endMark,
470                             Boolean.FALSE);
471                     state = new ParseBlockMappingFirstKey();
472                 } else if (anchor != null || tag != null) {
473                     // Empty scalars are allowed even if a tag or an anchor is
474                     // specified.
475                     event = new ScalarEvent(anchor, tag, new ImplicitTuple(implicit, false), "",
476                             startMark, endMark, (char) 0);
477                     state = states.pop();
478                 } else {
479                     String node;
480                     if (block) {
481                         node = "block";
482                     } else {
483                         node = "flow";
484                     }
485                     Token token = scanner.peekToken();
486                     throw new ParserException("while parsing a " + node + " node", startMark,
487                             "expected the node content, but found " + token.getTokenId(), token
488                                     .getStartMark());
489                 }
490             }
491         }
492         return event;
493     }

506     private class ParseBlockSequenceEntry implements Production {
507         public Event produce() {
508             if (scanner.checkToken(Token.ID.BlockEntry)) {
509                 BlockEntryToken token = (BlockEntryToken) scanner.getToken();
510                 if (!scanner.checkToken(Token.ID.BlockEntry, Token.ID.BlockEnd)) {
511                     states.push(new ParseBlockSequenceEntry());
512                     return new ParseBlockNode().produce();
513                 } else {
514                     state = new ParseBlockSequenceEntry();
515                     return processEmptyScalar(token.getEndMark());
516                 }
517             }
518             if (!scanner.checkToken(Token.ID.BlockEnd)) {
519                 Token token = scanner.peekToken();
520                 throw new ParserException("while parsing a block collection", marks.pop(),
521                         "expected <block end>, but found " + token.getTokenId(), token
522                                 .getStartMark());
523             }
524             Token token = scanner.getToken();
525             Event event = new SequenceEndEvent(token.getStartMark(), token.getEndMark());
526             state = states.pop();
527             marks.pop();
528             return event;
529         }
530     }

546             }

548             Event event = new SequenceEndEvent(token.getStartMark(), token.getEndMark());
549             state = states.pop();
550             return event;
551         }

562     private class ParseBlockMappingKey implements Production {
563         public Event produce() {
564             if (scanner.checkToken(Token.ID.Key)) {
565                 Token token = scanner.getToken();
566                 if (!scanner.checkToken(Token.ID.Key, Token.ID.Value, Token.ID.BlockEnd)) {
567                     states.push(new ParseBlockMappingValue());
568                     return parseBlockNodeOrIndentlessSequence();
569                 } else {
570                     state = new ParseBlockMappingValue();
571                     return processEmptyScalar(token.getEndMark());
572                 }
573             }
574             if (!scanner.checkToken(Token.ID.BlockEnd)) {
575                 Token token = scanner.peekToken();
576                 throw new ParserException("while parsing a block mapping", marks.pop(),
577                         "expected <block end>, but found " + token.getTokenId(), token
578                                 .getStartMark());
579             }
580             Token token = scanner.getToken();
581             Event event = new MappingEndEvent(token.getStartMark(), token.getEndMark());
582             state = states.pop();
583             marks.pop();
584             return event;
585         }
586     }

588     private class ParseBlockMappingValue implements Production {
589         public Event produce() {
590             if (scanner.checkToken(Token.ID.Value)) {
591                 Token token = scanner.getToken();
592                 if (!scanner.checkToken(Token.ID.Key, Token.ID.Value, Token.ID.BlockEnd)) {
593                     states.push(new ParseBlockMappingKey());
594                     return parseBlockNodeOrIndentlessSequence();
595                 } else {
596                     state = new ParseBlockMappingKey();
597                     return processEmptyScalar(token.getEndMark());
598                 }
599             }
600             state = new ParseBlockMappingKey();
601             Token token = scanner.peekToken();
602             return processEmptyScalar(token.getStartMark());
603         }
