136     /***
137      * <pre>
138      * A simple key is a key that is not denoted by the '?' indicator.
139      * Example of simple keys:
140      *   ---
141      *   block simple key: value
142      *   ? not a simple key:
143      *   : { flow simple key: value }
144      * We emit the KEY token before all keys, so when we find a potential
145      * simple key, we try to locate the corresponding ':' indicator.
146      * Simple keys should be limited to a single line and 1024 characters.
147      * 
148      * Can a simple key start at the current position? A simple key may
149      * start:
150      * - at the beginning of the line, not counting indentation spaces
151      *       (in block context),
152      * - after '{', '[', ',' (in the flow context),
153      * - after '?', ':', '-' (in the block context).
154      * In the block context, this flag also signifies if a block collection
155      * may start at the current position.
156      * </pre>
157      */
158     private boolean allowSimpleKey = true;
159 
160     /*
161      * Keep track of possible simple keys. This is a dictionary. The key is
162      * `flow_level`; there can be no more that one possible simple key for each
163      * level. The value is a SimpleKey record: (token_number, required, index,
164      * line, column, mark) A simple key may start with ALIAS, ANCHOR, TAG,
165      * SCALAR(flow), '[', or '{' tokens.
166      */
167     private Map<Integer, SimpleKey> possibleSimpleKeys;

169     public ScannerImpl(StreamReader reader) {
170         this.reader = reader;
171         this.tokens = new ArrayList<Token>(100);
172         this.indents = new ArrayStack<Integer>(10);
173         // the order in possibleSimpleKeys is kept for nextPossibleSimpleKey()

175         fetchStreamStart();// Add the STREAM-START token.

178     /***
179      * Check if the next token is one of the given types.
180      */
181     public boolean checkToken(Token.ID... choices) {
182         while (needMoreTokens()) {
183             fetchMoreTokens();
184         }
185         if (!this.tokens.isEmpty()) {
186             if (choices.length == 0) {
187                 return true;
188             }
189             Token first = this.tokens.get(0);
190             for (Token.ID id : choices) {
191                 if (first.getTokenId() == id) {
192                     return true;
193                 }
194             }
195         }
196         return false;
197     }

199     /***
200      * Return the next token, but do not delete if from the queue.
201      */
202     public Token peekToken() {
203         while (needMoreTokens()) {
204             fetchMoreTokens();
205         }
206         return this.tokens.get(0);
207     }

209     /***
210      * Return the next token.
211      */
212     public Token getToken() {
213         if (!this.tokens.isEmpty()) {
214             this.tokensTaken++;
215             return this.tokens.remove(0);
216         }
217         return null;
218     }

222     private boolean needMoreTokens() {
223         if (this.done) {
224             return false;
225         }
226         if (this.tokens.isEmpty()) {
227             return true;
228         }
229         // The current token may be a potential simple key, so we
230         // need to look further.
231         stalePossibleSimpleKeys();
232         return nextPossibleSimpleKey() == this.tokensTaken;
233     }
234 
235     private void fetchMoreTokens() {
236         // Eat whitespaces and comments until we reach the next token.
237         scanToNextToken();
238         // Remove obsolete possible simple keys.
239         stalePossibleSimpleKeys();
240         // Compare the current indentation and column. It may add some tokens
241         // and decrease the current indentation level.
242         unwindIndent(reader.getColumn());
243         // Peek the next character.
244         char ch = reader.peek();
245         switch (ch) {
246         case '\0':
247             // Is it the end of stream?
248             fetchStreamEnd();
249             return;
250         case '%':
251             // Is it a directive?
252             if (checkDirective()) {
253                 fetchDirective();
254                 return;
255             }
256             break;
257         case '-':
258             // Is it the document start?
259             if (checkDocumentStart()) {
260                 fetchDocumentStart();
261                 return;
262                 // Is it the block entry indicator?
263             } else if (checkBlockEntry()) {
264                 fetchBlockEntry();
265                 return;
266             }
267             break;
268         case '.':
269             // Is it the document end?
270             if (checkDocumentEnd()) {
271                 fetchDocumentEnd();
272                 return;
273             }
274             break;
275         // TODO support for BOM within a stream. (not implemented in PyYAML)
276         case '[':
277             // Is it the flow sequence start indicator?
278             fetchFlowSequenceStart();
279             return;
280         case '{':
281             // Is it the flow mapping start indicator?
282             fetchFlowMappingStart();
283             return;
284         case ']':
285             // Is it the flow sequence end indicator?
286             fetchFlowSequenceEnd();
287             return;
288         case '}':
289             // Is it the flow mapping end indicator?
290             fetchFlowMappingEnd();
291             return;
292         case ',':
293             // Is it the flow entry indicator?
294             fetchFlowEntry();
295             return;
296             // see block entry indicator above
297         case '?':
298             // Is it the key indicator?
299             if (checkKey()) {
300                 fetchKey();
301                 return;
302             }
303             break;
304         case ':':
305             // Is it the value indicator?
306             if (checkValue()) {
307                 fetchValue();
308                 return;
309             }
310             break;
311         case '*':
312             // Is it an alias?
313             fetchAlias();
314             return;
315         case '&':
316             // Is it an anchor?
317             fetchAnchor();
318             return;
319         case '!':
320             // Is it a tag?
321             fetchTag();
322             return;
323         case '|':
324             // Is it a literal scalar?
325             if (this.flowLevel == 0) {
326                 fetchLiteral();
327                 return;
328             }
329             break;
330         case '>':
331             // Is it a folded scalar?
332             if (this.flowLevel == 0) {
333                 fetchFolded();
334                 return;
335             }
336             break;
337         case '\'':
338             // Is it a single quoted scalar?
339             fetchSingle();
340             return;
341         case '"':
342             // Is it a double quoted scalar?
343             fetchDouble();
344             return;
345         }
346         // It must be a plain scalar then.
347         if (checkPlain()) {
348             fetchPlain();
349             return;
350         }
351         // No? It's an error. Let's produce a nice error message.
352         String chRepresentation = String.valueOf(ch);
353         for (Character s : ESCAPE_REPLACEMENTS.keySet()) {
354             String v = ESCAPE_REPLACEMENTS.get(s);
355             if (v.equals(chRepresentation)) {
356                 chRepresentation = "//" + s;// ' ' -> '\t'
357                 break;
358             }
359         }
360         throw new ScannerException("while scanning for the next token", null, "found character "
361                 + ch + "'" + chRepresentation + "' that cannot start any token", reader.getMark());
362     }

366     /***
367      * Return the number of the nearest possible simple key. Actually we don't
368      * need to loop through the whole dictionary.
369      */
370     private int nextPossibleSimpleKey() {
371         /*
372          * the implementation is not as in PyYAML. Because
373          * this.possibleSimpleKeys is ordered we can simply take the first key
374          */
375         Iterator<SimpleKey> iter = this.possibleSimpleKeys.values().iterator();
376         if (iter.hasNext()) {
377             SimpleKey key = iter.next();
378             return key.getTokenNumber();
379         }
380         return -1;
381     }

383     /***
384      * <pre>
385      * Remove entries that are no longer possible simple keys. According to
386      * the YAML specification, simple keys
387      * - should be limited to a single line,
388      * - should be no longer than 1024 characters.
389      * Disabling this procedure will allow simple keys of any length and
390      * height (may cause problems if indentation is broken though).
391      * </pre>
392      */
393     private void stalePossibleSimpleKeys() {
394         // use toRemove to avoid java.util.ConcurrentModificationException
395         Set<Integer> toRemove = null;
396         for (Integer level : this.possibleSimpleKeys.keySet()) {
397             SimpleKey key = this.possibleSimpleKeys.get(level);
398             if ((key.getLine() != reader.getLine()) || (reader.getIndex() - key.getIndex() > 1024)) {
399                 if (key.isRequired()) {
400                     throw new ScannerException("while scanning a simple key", key.getMark(),
401                             "could not found expected ':'", reader.getMark());
402                 } else {
403                     if (toRemove == null) {
404                         toRemove = new HashSet<Integer>();
405                     }
406                     toRemove.add(level);
407                 }
408             }
409         }
410         if (toRemove != null) {
411             for (Integer level : toRemove) {
412                 this.possibleSimpleKeys.remove(level);
413             }
414         }
415     }

417     /***
418      * The next token may start a simple key. We check if it's possible and save
419      * its position. This function is called for ALIAS, ANCHOR, TAG,
420      * SCALAR(flow), '[', and '{'.
421      */
422     private void savePossibleSimpleKey() {
423         // The next token may start a simple key. We check if it's possible
424         // and save its position. This function is called for
425         // ALIAS, ANCHOR, TAG, SCALAR(flow), '[', and '{'.
426 
427         // Check if a simple key is required at the current position.
428         boolean required = ((this.flowLevel == 0) && (this.indent == this.reader.getColumn()));
429 
430         if (allowSimpleKey || !required) {
431             // A simple key is required only if it is the first token in the
432             // current
433             // line. Therefore it is always allowed.
434         } else {
435             throw new YAMLException(
436                     "A simple key is required only if it is the first token in the current line");
437         }
438 
439         // The next token might be a simple key. Let's save it's number and
440         // position.
441         if (this.allowSimpleKey) {
442             removePossibleSimpleKey();
443             int tokenNumber = this.tokensTaken + this.tokens.size();
444             SimpleKey key = new SimpleKey(tokenNumber, required, reader.getIndex(), reader
445                     .getLine(), this.reader.getColumn(), this.reader.getMark());
446             this.possibleSimpleKeys.put(this.flowLevel, key);
447         }
448     }
449 
450     /***
451      * Remove the saved possible key position at the current flow level.
452      */
453     private void removePossibleSimpleKey() {
454         SimpleKey key = possibleSimpleKeys.get(flowLevel);
455         if (key != null && key.isRequired()) {
456             throw new ScannerException("while scanning a simple key", key.getMark(),
457                     "could not found expected ':'", reader.getMark());
458         }
459         possibleSimpleKeys.remove(flowLevel);
460     }

464     /***
465      * <pre>
466      * In flow context, tokens should respect indentation.
467      * Actually the condition should be `self.indent &gt;= column` according to
468      * the spec. But this condition will prohibit intuitively correct
469      * constructions such as
470      * key : {
471      * }
472      * </pre>
473      */
474     private void unwindIndent(int col) {
475         // In the flow context, indentation is ignored. We make the scanner less
476         // restrictive then specification requires.
477         if (this.flowLevel != 0) {
478             return;
479         }
480 
481         // In block context, we may need to issue the BLOCK-END tokens.
482         while (this.indent > col) {
483             Mark mark = reader.getMark();
484             this.indent = this.indents.pop();
485             this.tokens.add(new BlockEndToken(mark, mark));
486         }
487     }
488 
489     /***
490      * Check if we need to increase indentation.
491      */
492     private boolean addIndent(int column) {
493         if (this.indent < column) {
494             this.indents.push(this.indent);
495             this.indent = column;
496             return true;
497         }
498         return false;
499     }

516     private void fetchStreamEnd() {
517         // Set the current intendation to -1.
518         unwindIndent(-1);
519 
520         // Reset simple keys.
521         removePossibleSimpleKey();
522         this.allowSimpleKey = false;
523         this.possibleSimpleKeys.clear();
524 
525         // Read the token.
526         Mark mark = reader.getMark();
527 
528         // Add STREAM-END.
529         Token token = new StreamEndToken(mark, mark);
530         this.tokens.add(token);
531 
532         // The stream is finished.
533         this.done = true;
534     }

557     private void fetchDocumentIndicator(boolean isDocumentStart) {
558         // Set the current intendation to -1.
559         unwindIndent(-1);
560 
561         // Reset simple keys. Note that there could not be a block collection
562         // after '---'.
563         removePossibleSimpleKey();
564         this.allowSimpleKey = false;
565 
566         // Add DOCUMENT-START or DOCUMENT-END.
567         Mark startMark = reader.getMark();
568         reader.forward(3);
569         Mark endMark = reader.getMark();
570         Token token;
571         if (isDocumentStart) {
572             token = new DocumentStartToken(startMark, endMark);
573         } else {
574             token = new DocumentEndToken(startMark, endMark);
575         }
576         this.tokens.add(token);
577     }

587     private void fetchFlowCollectionStart(boolean isMappingStart) {
588         // '[' and '{' may start a simple key.
589         savePossibleSimpleKey();
590 
591         // Increase the flow level.
592         this.flowLevel++;
593 
594         // Simple keys are allowed after '[' and '{'.
595         this.allowSimpleKey = true;
596 
597         // Add FLOW-SEQUENCE-START or FLOW-MAPPING-START.
598         Mark startMark = reader.getMark();
599         reader.forward(1);
600         Mark endMark = reader.getMark();
601         Token token;
602         if (isMappingStart) {
603             token = new FlowMappingStartToken(startMark, endMark);
604         } else {
605             token = new FlowSequenceStartToken(startMark, endMark);
606         }
607         this.tokens.add(token);
608     }

618     private void fetchFlowCollectionEnd(boolean isMappingEnd) {
619         // Reset possible simple key on the current level.
620         removePossibleSimpleKey();
621 
622         // Decrease the flow level.
623         this.flowLevel--;
624 
625         // No simple keys after ']' or '}'.
626         this.allowSimpleKey = false;
627 
628         // Add FLOW-SEQUENCE-END or FLOW-MAPPING-END.
629         Mark startMark = reader.getMark();
630         reader.forward();
631         Mark endMark = reader.getMark();
632         Token token;
633         if (isMappingEnd) {
634             token = new FlowMappingEndToken(startMark, endMark);
635         } else {
636             token = new FlowSequenceEndToken(startMark, endMark);
637         }
638         this.tokens.add(token);
639     }

641     private void fetchFlowEntry() {
642         // Simple keys are allowed after ','.
643         this.allowSimpleKey = true;
644 
645         // Reset possible simple key on the current level.
646         removePossibleSimpleKey();
647 
648         // Add FLOW-ENTRY.
649         Mark startMark = reader.getMark();
650         reader.forward();
651         Mark endMark = reader.getMark();
652         Token token = new FlowEntryToken(startMark, endMark);
653         this.tokens.add(token);
654     }

656     private void fetchBlockEntry() {
657         // Block context needs additional checks.
658         if (this.flowLevel == 0) {
659             // Are we allowed to start a new entry?
660             if (!this.allowSimpleKey) {
661                 throw new ScannerException(null, null, "sequence entries are not allowed here",
662                         reader.getMark());
663             }
664 
665             // We may need to add BLOCK-SEQUENCE-START.
666             if (addIndent(this.reader.getColumn())) {
667                 Mark mark = reader.getMark();
668                 this.tokens.add(new BlockSequenceStartToken(mark, mark));
669             }
670         } else {
671             // It's an error for the block entry to occur in the flow
672             // context,but we let the parser detect this.
673         }
674         // Simple keys are allowed after '-'.
675         this.allowSimpleKey = true;
676 
677         // Reset possible simple key on the current level.
678         removePossibleSimpleKey();
679 
680         // Add BLOCK-ENTRY.
681         Mark startMark = reader.getMark();
682         reader.forward();
683         Mark endMark = reader.getMark();
684         Token token = new BlockEntryToken(startMark, endMark);
685         this.tokens.add(token);
686     }

688     private void fetchKey() {
689         // Block context needs additional checks.
690         if (this.flowLevel == 0) {
691             // Are we allowed to start a key (not necessary a simple)?
692             if (!this.allowSimpleKey) {
693                 throw new ScannerException(null, null, "mapping keys are not allowed here", reader
694                         .getMark());
695             }
696             // We may need to add BLOCK-MAPPING-START.
697             if (addIndent(this.reader.getColumn())) {
698                 Mark mark = reader.getMark();
699                 this.tokens.add(new BlockMappingStartToken(mark, mark));
700             }
701         }
702         // Simple keys are allowed after '?' in the block context.
703         this.allowSimpleKey = this.flowLevel == 0;
704 
705         // Reset possible simple key on the current level.
706         removePossibleSimpleKey();
707 
708         // Add KEY.
709         Mark startMark = reader.getMark();
710         reader.forward();
711         Mark endMark = reader.getMark();
712         Token token = new KeyToken(startMark, endMark);
713         this.tokens.add(token);
714     }

716     private void fetchValue() {
717         // Do we determine a simple key?
718         if (this.possibleSimpleKeys.keySet().contains(this.flowLevel)) {
719             // Add KEY.
720             SimpleKey key = this.possibleSimpleKeys.get(this.flowLevel);
721             this.possibleSimpleKeys.remove(this.flowLevel);
722             this.tokens.add(key.getTokenNumber() - this.tokensTaken, new KeyToken(key.getMark(),
723                     key.getMark()));
724 
725             // If this key starts a new block mapping, we need to add
726             // BLOCK-MAPPING-START.
727             if (this.flowLevel == 0) {
728                 if (addIndent(key.getColumn())) {
729                     this.tokens.add(key.getTokenNumber() - this.tokensTaken,
730                             new BlockMappingStartToken(key.getMark(), key.getMark()));
731                 }
732             }
733             // There cannot be two simple keys one after another.
734             this.allowSimpleKey = false;
735 
736         } else {// It must be a part of a complex key.
737             // Block context needs additional checks.Do we really need them?
738             // They
739             // will be catched by the parser anyway.)
740             if (this.flowLevel == 0) {
741 
742                 // We are allowed to start a complex value if and only if we can
743                 // start a simple key.
744                 if (!this.allowSimpleKey) {
745                     throw new ScannerException(null, null, "mapping values are not allowed here",
746                             reader.getMark());
747                 }
748             }
749 
750             // If this value starts a new block mapping, we need to add
751             // BLOCK-MAPPING-START. It will be detected as an error later by
752             // the parser.
753             if (flowLevel == 0) {
754                 if (addIndent(reader.getColumn())) {
755                     Mark mark = reader.getMark();
756                     this.tokens.add(new BlockMappingStartToken(mark, mark));
757                 }
758             }
759 
760             // Simple keys are allowed after ':' in the block context.
761             allowSimpleKey = (flowLevel == 0);
762 
763             // Reset possible simple key on the current level.
764             removePossibleSimpleKey();
765         }
766         // Add VALUE.
767         Mark startMark = reader.getMark();
768         reader.forward();
769         Mark endMark = reader.getMark();
770         Token token = new ValueToken(startMark, endMark);
771         this.tokens.add(token);
772     }

917     private boolean checkPlain() {
918         /***
919          * <pre>
920          * A plain scalar may start with any non-space character except:
921          *   '-', '?', ':', ',', '[', ']', '{', '}',
922          *   '#', '&amp;', '*', '!', '|', '&gt;', '\'', '\&quot;',

926          *   '-', '?', ':'

928          *

942     /***
943      * <pre>
944      * We ignore spaces, line breaks and comments.
945      * If we find a line break in the block context, we set the flag
946      * `allow_simple_key` on.
947      * The byte order mark is stripped if it's the first character in the
948      * stream. We do not yet support BOM inside the stream as the
949      * specification requires. Any such mark will be considered as a part
950      * of the document.
951      * TODO: We need to make tab handling rules more sane. A good rule is
952      *   Tabs cannot precede tokens
953      *   BLOCK-SEQUENCE-START, BLOCK-MAPPING-START, BLOCK-END,
954      *   KEY(block), VALUE(block), BLOCK-ENTRY
955      * So the checking code is
956      *   if &lt;TAB&gt;:
957      *       self.allow_simple_keys = False
958      * We also need to add the check for `allow_simple_keys == True` to
959      * `unwind_indent` before issuing BLOCK-END.
960      * Scanners for block, flow, and plain scalars need to be modified.
961      * </pre>
962      */
963     private void scanToNextToken() {
964         if (reader.getIndex() == 0 && reader.peek() == '\uFEFF') {
965             reader.forward();
966         }
967         boolean found = false;
968         while (!found) {
969             while (reader.peek() == ' ') {
970                 reader.forward();
971             }
972             if (reader.peek() == '#') {
973                 while (Constant.NULL_OR_LINEBR.hasNo(reader.peek())) {
974                     reader.forward();
975                 }
976             }
977             if (scanLineBreak().length() != 0) {
978                 if (this.flowLevel == 0) {
979                     this.allowSimpleKey = true;
980                 }
981             } else {
982                 found = true;
983             }
984         }
985     }

988     private Token scanDirective() {
989         // See the specification for details.
990         Mark startMark = reader.getMark();
991         Mark endMark;
992         reader.forward();
993         String name = scanDirectiveName(startMark);
994         List<?> value = null;
995         if ("YAML".equals(name)) {

997             endMark = reader.getMark();
998         } else if ("TAG".equals(name)) {

1000             endMark = reader.getMark();
1001         } else {

1004                 reader.forward();
1005             }
1006         }
1007         scanDirectiveIgnoredLine(startMark);
1008         return new DirectiveToken(name, value, startMark, endMark);
1009     }

1132     /***
1133      * <pre>
1134      * The specification does not restrict characters for anchors and
1135      * aliases. This may lead to problems, for instance, the document:
1136      *   [ *alias, value ]
1137      * can be interpreted in two ways, as
1138      *   [ &quot;value&quot; ]
1139      * and
1140      *   [ *alias , &quot;value&quot; ]
1141      * Therefore we restrict aliases to numbers and ASCII letters.
1142      * </pre>
1143      */

1145         Mark startMark = reader.getMark();

1148         reader.forward();
1149         int length = 0;
1150         char ch = reader.peek(length);
1151         while (Constant.ALPHA.indexOf(ch) != -1) {
1152             length++;
1153             ch = reader.peek(length);
1154         }

1157                     "expected alphabetic or numeric character, but found but found " + ch, reader
1158                             .getMark());
1159         }
1160         String value = reader.prefix(length);
1161         reader.forward(length);
1162         ch = reader.peek();

1167         }
1168         Mark endMark = reader.getMark();
1169         Token tok;
1170         if (isAnchor) {
1171             tok = new AnchorToken(value, startMark, endMark);
1172         } else {
1173             tok = new AliasToken(value, startMark, endMark);

1178     private Token scanTag() {
1179         // See the specification for details.
1180         Mark startMark = reader.getMark();
1181         char ch = reader.peek(1);
1182         String handle = null;
1183         String suffix = null;
1184         if (ch == '<') {
1185             reader.forward(2);
1186             suffix = scanTagUri("tag", startMark);
1187             if (reader.peek() != '>') {
1188                 throw new ScannerException("while scanning a tag", startMark,
1189                         "expected '>', but found '" + reader.peek() + "' (" + ((int) reader.peek())
1190                                 + ")", reader.getMark());
1191             }
1192             reader.forward();
1193         } else if (Constant.NULL_BL_T_LINEBR.has(ch)) {
1194             suffix = "!";
1195             reader.forward();
1196         } else {
1197             int length = 1;
1198             boolean useHandle = false;
1199             while (Constant.NULL_BL_LINEBR.hasNo(ch)) {
1200                 if (ch == '!') {
1201                     useHandle = true;
1202                     break;
1203                 }
1204                 length++;
1205                 ch = reader.peek(length);
1206             }
1207             handle = "!";
1208             if (useHandle) {
1209                 handle = scanTagHandle("tag", startMark);
1210             } else {
1211                 handle = "!";
1212                 reader.forward();
1213             }
1214             suffix = scanTagUri("tag", startMark);
1215         }
1216         ch = reader.peek();
1217         if (Constant.NULL_BL_LINEBR.hasNo(ch)) {
1218             throw new ScannerException("while scanning a tag", startMark,
1219                     "expected ' ', but found '" + ch + "' (" + ((int) ch) + ")", reader.getMark());
1220         }
1221         TagTuple value = new TagTuple(handle, suffix);
1222         Mark endMark = reader.getMark();
1223         return new TagToken(value, startMark, endMark);

1226     private Token scanBlockScalar(char style) {
1227         // See the specification for details.
1228         boolean folded;
1229         if (style == '>') {
1230             folded = true;
1231         } else {
1232             folded = false;
1233         }
1234         StringBuilder chunks = new StringBuilder();
1235         Mark startMark = reader.getMark();
1236         // Scan the header.
1237         reader.forward();
1238         Chomping chompi = scanBlockScalarIndicators(startMark);
1239         int increment = chompi.getIncrement();
1240         scanBlockScalarIgnoredLine(startMark);
1241 
1242         // Determine the indentation level and go to the first non-empty line.
1243         int minIndent = this.indent + 1;
1244         if (minIndent < 1) {
1245             minIndent = 1;
1246         }
1247         String breaks = null;
1248         int maxIndent = 0;
1249         int indent = 0;
1250         Mark endMark;
1251         if (increment == -1) {
1252             Object[] brme = scanBlockScalarIndentation();
1253             breaks = (String) brme[0];
1254             maxIndent = ((Integer) brme[1]).intValue();
1255             endMark = (Mark) brme[2];
1256             indent = Math.max(minIndent, maxIndent);
1257         } else {
1258             indent = minIndent + increment - 1;
1259             Object[] brme = scanBlockScalarBreaks(indent);
1260             breaks = (String) brme[0];
1261             endMark = (Mark) brme[1];
1262         }
1263 
1264         String lineBreak = "";
1265 
1266         // Scan the inner part of the block scalar.
1267         while (this.reader.getColumn() == indent && reader.peek() != '\0') {
1268             chunks.append(breaks);
1269             boolean leadingNonSpace = " \t".indexOf(reader.peek()) == -1;
1270             int length = 0;
1271             while (Constant.NULL_OR_LINEBR.hasNo(reader.peek(length))) {
1272                 length++;
1273             }
1274             chunks.append(reader.prefix(length));
1275             reader.forward(length);
1276             lineBreak = scanLineBreak();
1277             Object[] brme = scanBlockScalarBreaks(indent);
1278             breaks = (String) brme[0];
1279             endMark = (Mark) brme[1];
1280             if (this.reader.getColumn() == indent && reader.peek() != '\0') {
1281 
1282                 // Unfortunately, folding rules are ambiguous.
1283                 //
1284                 // This is the folding according to the specification:
1285                 if (folded && "\n".equals(lineBreak) && leadingNonSpace
1286                         && " \t".indexOf(reader.peek()) == -1) {
1287                     if (breaks.length() == 0) {
1288                         chunks.append(" ");
1289                     }
1290                 } else {
1291                     chunks.append(lineBreak);
1292                 }
1293                 // Clark Evans's interpretation (also in the spec examples) not
1294                 // imported from PyYAML
1295             } else {
1296                 break;
1297             }
1298         }
1299         // Chomp the tail.
1300         if (chompi.chompTailIsNotFalse()) {
1301             chunks.append(lineBreak);
1302         }
1303         if (chompi.chompTailIsTrue()) {
1304             chunks.append(breaks);
1305         }
1306         // We are done.
1307         return new ScalarToken(chunks.toString(), false, startMark, endMark, style);
1308     }

1310     private Chomping scanBlockScalarIndicators(Mark startMark) {
1311         // See the specification for details.
1312         Boolean chomping = null;
1313         int increment = -1;
1314         char ch = reader.peek();
1315         if (ch == '-' || ch == '+') {
1316             if (ch == '+') {
1317                 chomping = Boolean.TRUE;
1318             } else {
1319                 chomping = Boolean.FALSE;
1320             }
1321             reader.forward();
1322             ch = reader.peek();
1323             if (Character.isDigit(ch)) {
1324                 increment = Integer.parseInt(String.valueOf(ch));
1325                 if (increment == 0) {
1326                     throw new ScannerException("while scanning a block scalar", startMark,
1327                             "expected indentation indicator in the range 1-9, but found 0", reader
1328                                     .getMark());
1329                 }
1330                 reader.forward();
1331             }
1332         } else if (Character.isDigit(ch)) {
1333             increment = Integer.parseInt(String.valueOf(ch));
1334             if (increment == 0) {
1335                 throw new ScannerException("while scanning a block scalar", startMark,
1336                         "expected indentation indicator in the range 1-9, but found 0", reader
1337                                 .getMark());
1338             }
1339             reader.forward();
1340             ch = reader.peek();
1341             if (ch == '-' || ch == '+') {
1342                 if (ch == '+') {
1343                     chomping = Boolean.TRUE;
1344                 } else {
1345                     chomping = Boolean.FALSE;
1346                 }
1347                 reader.forward();
1348             }
1349         }
1350         ch = reader.peek();
1351         if (Constant.NULL_BL_LINEBR.hasNo(ch)) {
1352             throw new ScannerException("while scanning a block scalar", startMark,
1353                     "expected chomping or indentation indicators, but found " + ch, reader
1354                             .getMark());
1355         }
1356         return new Chomping(chomping, increment);

1716         // encoded in UTF-8, and then each octet is written as a

1743     private String scanLineBreak() {
1744         // Transforms:
1745         // '\r\n' : '\n'
1746         // '\r' : '\n'
1747         // '\n' : '\n'
1748         // '\x85' : '\n'
1749         // default : ''
1750         char ch = reader.peek();
1751         if ("\r\n\u0085".indexOf(ch) != -1) {
1752             if ("\r\n".equals(reader.prefix(2))) {
1753                 reader.forward(2);
1754             } else {
1755                 reader.forward();
1756             }
1757             return "\n";
1758         } else if ("\u2028\u2029".indexOf(ch) != -1) {
1759             reader.forward();
1760             return String.valueOf(ch);
1761         }
1762         return "";
