137     /***
138      * Create Yaml instance. It is safe to create a few instances and use them
139      * in different Threads.
140      * 
141      * @param constructor
142      *            BaseConstructor to construct incoming documents
143      * @param representer
144      *            Representer to emit outgoing objects
145      * @param options
146      *            DumperOptions to configure outgoing objects
147      * @param resolver
148      *            Resolver to detect implicit type
149      */
150     public Yaml(BaseConstructor constructor, Representer representer, DumperOptions options,
151             Resolver resolver) {
152         if(!constructor.isExplicitPropertyUtils()) {
153             constructor.setPropertyUtils(representer.getPropertyUtils());
154         } else if(!representer.isExplicitPropertyUtils()) {
155             representer.setPropertyUtils(constructor.getPropertyUtils());
156         }
157         this.constructor = constructor;
158         representer.setDefaultFlowStyle(options.getDefaultFlowStyle());
159         representer.setDefaultScalarStyle(options.getDefaultScalarStyle());
160         representer.getPropertyUtils().setAllowReadOnlyProperties(
161                 options.isAllowReadOnlyProperties());
162         this.representer = representer;
163         this.options = options;
164         this.resolver = resolver;
165         this.name = "Yaml:" + System.identityHashCode(this);
166     }

208     /***
209      * Serialize a sequence of Java objects into a YAML stream.
210      * 
211      * @param data
212      *            Iterator with Objects
213      * @param output
214      *            stream to write to
215      */
216     public void dumpAll(Iterator<? extends Object> data, Writer output) {
217         Serializer s = new Serializer(new Emitter(output, options), resolver, options);
218         try {
219             s.open();
220             while (data.hasNext()) {
221                 representer.represent(s, data.next());
222             }
223             s.close();
224         } catch (java.io.IOException e) {
225             throw new YAMLException(e);
226         }
227     }

277         Composer composer = new Composer(new ParserImpl(new StreamReader(yaml)), resolver);
278         constructor.setComposer(composer);

280             public boolean hasNext() {

290             }

447     /***
448      * Parse a YAML stream and produce parsing events.
449      * 
450      * @param yaml
451      *            YAML document(s)
452      * @return parsed events
453      */
454     public Iterable<Event> parse(Reader yaml) {
455         final Parser parser = new ParserImpl(new StreamReader(yaml));
456         Iterator<Event> result = new Iterator<Event>() {
457             public boolean hasNext() {
458                 return parser.peekEvent() != null;
459             }
460 
461             public Event next() {
462                 return parser.getEvent();
463             }
464 
465             public void remove() {
466                 throw new UnsupportedOperationException();
467             }
468         };
469         return new EventIterable(result);
470     }
