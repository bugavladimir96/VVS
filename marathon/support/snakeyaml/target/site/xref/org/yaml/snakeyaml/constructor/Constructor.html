48  /***
49   * Construct a custom Java instance.
50   */
51  public class Constructor extends SafeConstructor {
52      private final Map<Tag, Class<? extends Object>> typeTags;
53      private final Map<Class<? extends Object>, TypeDescription> typeDefinitions;
54  
55      public Constructor() {
56          this(Object.class);
57      }

59      /***
60       * Create Constructor for the specified class as the root.
61       * 
62       * @param theRoot
63       *            - the class (usually JavaBean) to be constructed
64       */
65      public Constructor(Class<? extends Object> theRoot) {
66          if (theRoot == null) {
67              throw new NullPointerException("Root type must be provided.");
68          }
69          this.yamlConstructors.put(null, new ConstructYamlObject());
70          if (!Object.class.equals(theRoot)) {
71              rootTag = new Tag(theRoot);
72          }
73          typeTags = new HashMap<Tag, Class<? extends Object>>();
74          typeDefinitions = new HashMap<Class<? extends Object>, TypeDescription>();
75          yamlClassConstructors.put(NodeId.scalar, new ConstructScalar());
76          yamlClassConstructors.put(NodeId.mapping, new ConstructMapping());
77          yamlClassConstructors.put(NodeId.sequence, new ConstructSequence());
78      }

80      /***
81       * Create Constructor for a class which does not have to be in the classpath
82       * or for a definition from a Spring ApplicationContext.
83       * 
84       * @param theRoot
85       *            fully qualified class name of the root class (usually
86       *            JavaBean)
87       * @throws ClassNotFoundException
88       */
89      public Constructor(String theRoot) throws ClassNotFoundException {
90          this(Class.forName(check(theRoot)));
91      }

93      private static final String check(String s) {
94          if (s == null) {
95              throw new NullPointerException("Root type must be provided.");
96          }
97          if (s.trim().length() == 0) {
98              throw new YAMLException("Root type must be provided.");
99          }
100         return s;
101     }

103     /***
104      * Make YAML aware how to parse a custom Class. If there is no root Class
105      * assigned in constructor then the 'root' property of this definition is
106      * respected.
107      * 
108      * @param definition
109      *            to be added to the Constructor
110      * @return the previous value associated with <tt>definition</tt>, or
111      *         <tt>null</tt> if there was no mapping for <tt>definition</tt>.
112      */
113     public TypeDescription addTypeDescription(TypeDescription definition) {
114         if (definition == null) {
115             throw new NullPointerException("TypeDescription is required.");
116         }
117         if (rootTag == null && definition.isRoot()) {
118             rootTag = new Tag(definition.getType());
119         }
120         Tag tag = definition.getTag();
121         typeTags.put(tag, definition.getType());
122         return typeDefinitions.put(definition.getType(), definition);
123     }

131         /***
132          * Construct JavaBean. If type safe collections are used please look at
133          * <code>TypeDescription</code>.
134          * 
135          * @param node
136          *            node where the keys are property names (they can only be
137          *            <code>String</code>s) and values are objects to be created
138          * @return constructed JavaBean
139          */
140         public Object construct(Node node) {
141             MappingNode mnode = (MappingNode) node;
142             if (Properties.class.isAssignableFrom(node.getType())) {
143                 Properties properties = new Properties();
144                 if (!node.isTwoStepsConstruction()) {
145                     constructMapping2ndStep(mnode, (Map<Object, Object>) properties);
146                 } else {
147                     throw new YAMLException("Properties must not be recursive.");
148                 }
149                 return properties;
150             } else if (SortedMap.class.isAssignableFrom(node.getType())) {
151                 SortedMap<Object, Object> map = new TreeMap<Object, Object>();
152                 if (!node.isTwoStepsConstruction()) {
153                     constructMapping2ndStep(mnode, map);
154                 }
155                 return map;
156             } else if (Map.class.isAssignableFrom(node.getType())) {
157                 if (node.isTwoStepsConstruction()) {
158                     return createDefaultMap();
159                 } else {
160                     return constructMapping(mnode);
161                 }
162             } else if (SortedSet.class.isAssignableFrom(node.getType())) {
163                 SortedSet<Object> set = new TreeSet<Object>();
164                 // XXX why this is not used ?
165                 // if (!node.isTwoStepsConstruction()) {
166                 constructSet2ndStep(mnode, set);
167                 // }
168                 return set;
169             } else if (Collection.class.isAssignableFrom(node.getType())) {
170                 if (node.isTwoStepsConstruction()) {
171                     return createDefaultSet();
172                 } else {
173                     return constructSet(mnode);
174                 }
175             } else {
176                 if (node.isTwoStepsConstruction()) {
177                     return createEmptyJavaBean(mnode);
178                 } else {
179                     return constructJavaBean2ndStep(mnode, createEmptyJavaBean(mnode));
180                 }
181             }
182         }

184         @SuppressWarnings("unchecked")
185         public void construct2ndStep(Node node, Object object) {
186             if (Map.class.isAssignableFrom(node.getType())) {
187                 constructMapping2ndStep((MappingNode) node, (Map<Object, Object>) object);
188             } else if (Set.class.isAssignableFrom(node.getType())) {
189                 constructSet2ndStep((MappingNode) node, (Set<Object>) object);
190             } else {
191                 constructJavaBean2ndStep((MappingNode) node, object);
192             }
193         }

195         private Object createEmptyJavaBean(MappingNode node) {
196             try {
197                 Class<? extends Object> type = node.getType();
198                 if (Modifier.isAbstract(type.getModifiers())) {
199                     node.setType(getClassForNode(node));
200                 }
201                 /***
202                  * Using only default constructor. Everything else will be
203                  * initialized on 2nd step. If we do here some partial
204                  * initialization, how do we then track what need to be done on
205                  * 2nd step? I think it is better to get only object here (to
206                  * have it as reference for recursion) and do all other thing on
207                  * 2nd step.
208                  */
209                 java.lang.reflect.Constructor<?> c = node.getType().getDeclaredConstructor();
210                 c.setAccessible(true);
211                 return c.newInstance();
212             } catch (Exception e) {
213                 throw new YAMLException(e);
214             }
215         }

217         @SuppressWarnings("unchecked")
218         private Object constructJavaBean2ndStep(MappingNode node, Object object) {
219             Class<? extends Object> beanType = node.getType();
220             List<NodeTuple> nodeValue = (List<NodeTuple>) node.getValue();
221             for (NodeTuple tuple : nodeValue) {
222                 ScalarNode keyNode;
223                 if (tuple.getKeyNode() instanceof ScalarNode) {
224                     // key must be scalar
225                     keyNode = (ScalarNode) tuple.getKeyNode();
226                 } else {
227                     throw new YAMLException("Keys must be scalars but found: " + tuple.getKeyNode());
228                 }
229                 Node valueNode = tuple.getValueNode();
230                 // keys can only be Strings
231                 keyNode.setType(String.class);
232                 String key = (String) constructObject(keyNode);
233                 try {
234                     Property property = getProperty(beanType, key);
235                     valueNode.setType(property.getType());
236                     TypeDescription memberDescription = typeDefinitions.get(beanType);
237                     boolean typeDetected = false;
238                     if (memberDescription != null) {
239                         switch (valueNode.getNodeId()) {
240                         case sequence:
241                             SequenceNode snode = (SequenceNode) valueNode;
242                             Class<? extends Object> memberType = memberDescription
243                                     .getListPropertyType(key);
244                             if (memberType != null) {
245                                 snode.setListType(memberType);
246                                 typeDetected = true;
247                             } else if (property.getType().isArray()) {
248                                 snode.setListType(property.getType().getComponentType());
249                                 typeDetected = true;
250                             }
251                             break;
252                         case mapping:
253                             MappingNode mnode = (MappingNode) valueNode;
254                             Class<? extends Object> keyType = memberDescription.getMapKeyType(key);
255                             if (keyType != null) {
256                                 mnode.setKeyType(keyType);
257                                 mnode.setValueType(memberDescription.getMapValueType(key));
258                                 typeDetected = true;
259                             }
260                             break;
261                         }
262                     }
263                     if (!typeDetected && valueNode.getNodeId() != NodeId.scalar) {
264                         // only if there is no explicit TypeDescription
265                         Class[] arguments = property.getActualTypeArguments();
266                         if (arguments != null) {
267                             // type safe (generic) collection may contain the
268                             // proper class
269                             if (valueNode.getNodeId() == NodeId.sequence) {
270                                 Class t = arguments[0];
271                                 SequenceNode snode = (SequenceNode) valueNode;
272                                 snode.setListType(t);
273                             } else if (valueNode.getTag().equals(Tag.SET)) {
274                                 Class t = arguments[0];
275                                 MappingNode mnode = (MappingNode) valueNode;
276                                 mnode.setKeyType(t);
277                                 mnode.setUseClassConstructor(true);
278                             } else if (valueNode.getNodeId() == NodeId.mapping) {
279                                 Class ketType = arguments[0];
280                                 Class valueType = arguments[1];
281                                 MappingNode mnode = (MappingNode) valueNode;
282                                 mnode.setKeyType(ketType);
283                                 mnode.setValueType(valueType);
284                                 mnode.setUseClassConstructor(true);
285                             }
286                         }
287                     }
288                     Object value = constructObject(valueNode);
289                     property.set(object, value);
290                 } catch (Exception e) {
291                     throw new YAMLException("Cannot create property=" + key + " for JavaBean="
292                             + object + "; " + e.getMessage(), e);
293                 }
294             }
295             return object;
296         }

314             Class cl = getClassForNode(node);
315             node.setType(cl);
316             // call the constructor as if the runtime class is defined
317             Construct constructor = yamlClassConstructors.get(node.getNodeId());

322             Object result = null;

325             } catch (Exception e) {
326                 throw new ConstructorException(null, null, "Can't construct a java object for "
327                         + node.getTag() + "; exception=" + e.getMessage(), node.getStartMark(), e);
328             }
329             return result;

343     /***
344      * Construct scalar instance when the runtime class is known. Recursive
345      * structures are not supported.
346      */
347     protected class ConstructScalar extends AbstractConstruct {
348         @SuppressWarnings("unchecked")
349         public Object construct(Node nnode) {
350             ScalarNode node = (ScalarNode) nnode;
351             Class type = node.getType();
352             Object result;
353             if (type.isPrimitive() || type == String.class || Number.class.isAssignableFrom(type)
354                     || type == Boolean.class || Date.class.isAssignableFrom(type)
355                     || type == Character.class || type == BigInteger.class
356                     || type == BigDecimal.class || Enum.class.isAssignableFrom(type)
357                     || Tag.BINARY.equals(node.getTag()) || Calendar.class.isAssignableFrom(type)) {
358                 // standard classes created directly
359                 result = constructStandardJavaInstance(type, node);
360             } else {
361                 // there must be only 1 constructor with 1 argument
362                 java.lang.reflect.Constructor[] javaConstructors = type.getConstructors();
363                 int oneArgCount = 0;
364                 java.lang.reflect.Constructor javaConstructor = null;
365                 for (java.lang.reflect.Constructor c : javaConstructors) {
366                     if (c.getParameterTypes().length == 1) {
367                         oneArgCount++;
368                         javaConstructor = c;
369                     }
370                 }
371                 Object argument;
372                 if (javaConstructor == null) {
373                     throw new YAMLException("No single argument constructor found for " + type);
374                 } else if (oneArgCount == 1) {
375                     argument = constructStandardJavaInstance(
376                             javaConstructor.getParameterTypes()[0], node);
377                 } else {
378                     // TODO it should be possible to use implicit types instead
379                     // of forcing String. Resolver must be available here to
380                     // obtain the implicit tag. Then we can set the tag and call
381                     // callConstructor(node) to create the argument instance.
382                     // On the other hand it may be safer to require a custom
383                     // constructor to avoid guessing the argument class
384                     argument = constructScalar(node);
385                     try {
386                         javaConstructor = type.getConstructor(String.class);
387                     } catch (Exception e) {
388                         throw new ConstructorException(null, null,
389                                 "Can't construct a java object for scalar " + node.getTag()
390                                         + "; No String constructor found. Exception="
391                                         + e.getMessage(), node.getStartMark(), e);
392                     }
393                 }
394                 try {
395                     result = javaConstructor.newInstance(argument);
396                 } catch (Exception e) {
397                     throw new ConstructorException(null, null,
398                             "Can't construct a java object for scalar " + node.getTag()
399                                     + "; exception=" + e.getMessage(), node.getStartMark(), e);
400                 }
401             }
402             return result;
403         }
404 
405         @SuppressWarnings("unchecked")
406         private Object constructStandardJavaInstance(Class type, ScalarNode node) {
407             Object result;
408             if (type == String.class) {
409                 Construct stringConstructor = yamlConstructors.get(Tag.STR);
410                 result = stringConstructor.construct((ScalarNode) node);
411             } else if (type == Boolean.class || type == Boolean.TYPE) {
412                 Construct boolConstructor = yamlConstructors.get(Tag.BOOL);
413                 result = boolConstructor.construct((ScalarNode) node);
414             } else if (type == Character.class || type == Character.TYPE) {
415                 Construct charConstructor = yamlConstructors.get(Tag.STR);
416                 String ch = (String) charConstructor.construct((ScalarNode) node);
417                 if (ch.length() == 0) {
418                     result = null;
419                 } else if (ch.length() != 1) {
420                     throw new YAMLException("Invalid node Character: '" + ch + "'; length: "
421                             + ch.length());
422                 } else {
423                     result = new Character(ch.charAt(0));
424                 }
425             } else if (Date.class.isAssignableFrom(type)) {
426                 Construct dateConstructor = yamlConstructors.get(Tag.TIMESTAMP);
427                 Date date = (Date) dateConstructor.construct((ScalarNode) node);
428                 if (type == Date.class) {
429                     result = date;
430                 } else {
431                     try {
432                         java.lang.reflect.Constructor<?> constr = type.getConstructor(long.class);
433                         result = constr.newInstance(date.getTime());
434                     } catch (Exception e) {
435                         throw new YAMLException("Cannot construct: '" + type + "'");
436                     }
437                 }
438             } else if (type == Float.class || type == Double.class || type == Float.TYPE
439                     || type == Double.TYPE || type == BigDecimal.class) {
440                 if (type == BigDecimal.class) {
441                     result = new BigDecimal(node.getValue());
442                 } else {
443                     Construct doubleConstructor = yamlConstructors.get(Tag.FLOAT);
444                     result = doubleConstructor.construct(node);
445                     if (type == Float.class || type == Float.TYPE) {
446                         result = new Float((Double) result);
447                     }
448                 }
449             } else if (type == Byte.class || type == Short.class || type == Integer.class
450                     || type == Long.class || type == BigInteger.class || type == Byte.TYPE
451                     || type == Short.TYPE || type == Integer.TYPE || type == Long.TYPE) {
452                 Construct intConstructor = yamlConstructors.get(Tag.INT);
453                 result = intConstructor.construct(node);
454                 if (type == Byte.class || type == Byte.TYPE) {
455                     result = new Byte(result.toString());
456                 } else if (type == Short.class || type == Short.TYPE) {
457                     result = new Short(result.toString());
458                 } else if (type == Integer.class || type == Integer.TYPE) {
459                     result = new Integer(result.toString());
460                 } else if (type == Long.class || type == Long.TYPE) {
461                     result = new Long(result.toString());
462                 } else {
463                     // only BigInteger left
464                     result = new BigInteger(result.toString());
465                 }
466             } else if (Enum.class.isAssignableFrom(type)) {
467                 String enumValueName = node.getValue();
468                 try {
469                     result = Enum.valueOf(type, enumValueName);
470                 } catch (Exception ex) {
471                     throw new YAMLException("Unable to find enum value '" + enumValueName
472                             + "' for enum class: " + type.getName());
473                 }
474             } else if (Calendar.class.isAssignableFrom(type)) {
475                 ConstructYamlTimestamp contr = new ConstructYamlTimestamp();
476                 contr.construct(node);
477                 result = contr.getCalendar();
478             } else {
479                 throw new YAMLException("Unsupported class: " + type);
480             }
481             return result;
482         }

485     /***
486      * Construct sequence (List, Array, or immutable object) when the runtime
487      * class is known. TODO make protected
488      */
489     private class ConstructSequence implements Construct {
490         @SuppressWarnings("unchecked")
491         public Object construct(Node node) {
492             SequenceNode snode = (SequenceNode) node;
493             if (Set.class.isAssignableFrom(node.getType())) {
494                 if (node.isTwoStepsConstruction()) {
495                     throw new YAMLException("Set cannot be recursive.");
496                 } else {
497                     return constructSet(snode);
498                 }
499             } else if (Collection.class.isAssignableFrom(node.getType())) {
500                 if (node.isTwoStepsConstruction()) {
501                     return createDefaultList(snode.getValue().size());
502                 } else {
503                     return constructSequence(snode);
504                 }
505             } else if (node.getType().isArray()) {
506                 if (node.isTwoStepsConstruction()) {
507                     return createArray(node.getType(), snode.getValue().size());
508                 } else {
509                     return constructArray(snode);
510                 }
511             } else {
512                 // create immutable object
513                 List<java.lang.reflect.Constructor> possibleConstructors = new ArrayList<java.lang.reflect.Constructor>(
514                         snode.getValue().size());
515                 for (java.lang.reflect.Constructor constructor : node.getType().getConstructors()) {
516                     if (snode.getValue().size() == constructor.getParameterTypes().length) {
517                         possibleConstructors.add(constructor);
518                     }
519                 }
520                 if (!possibleConstructors.isEmpty()) {
521                     if (possibleConstructors.size() == 1) {
522                         Object[] argumentList = new Object[snode.getValue().size()];
523                         java.lang.reflect.Constructor c = possibleConstructors.get(0);
524                         int index = 0;
525                         for (Node argumentNode : snode.getValue()) {
526                             Class type = c.getParameterTypes()[index];
527                             // set runtime classes for arguments
528                             argumentNode.setType(type);
529                             argumentList[index++] = constructObject(argumentNode);
530                         }
531 
532                         try {
533                             return c.newInstance(argumentList);
534                         } catch (Exception e) {
535                             throw new YAMLException(e);
536                         }
537                     }
538 
539                     // use BaseConstructor
540                     List<Object> argumentList = (List<Object>) constructSequence(snode);
541                     Class[] parameterTypes = new Class[argumentList.size()];
542                     int index = 0;
543                     for (Object parameter : argumentList) {
544                         parameterTypes[index] = parameter.getClass();
545                         index++;
546                     }
547 
548                     for (java.lang.reflect.Constructor c : possibleConstructors) {
549                         Class[] argTypes = c.getParameterTypes();
550                         boolean foundConstructor = true;
551                         for (int i = 0; i < argTypes.length; i++) {
552                             if (!wrapIfPrimitive(argTypes[i]).isAssignableFrom(parameterTypes[i])) {
553                                 foundConstructor = false;
554                                 break;
555                             }
556                         }
557                         if (foundConstructor) {
558                             try {
559                                 return c.newInstance(argumentList.toArray());
560                             } catch (Exception e) {
561                                 throw new YAMLException(e);
562                             }
563                         }
564                     }
565                 }
566                 throw new YAMLException("No suitable constructor with "
567                         + String.valueOf(snode.getValue().size()) + " arguments found for "
568                         + node.getType());
569 
570             }
571         }

573         private final Class<? extends Object> wrapIfPrimitive(Class<?> clazz) {
574             if (!clazz.isPrimitive()) {
575                 return clazz;
576             }
577             if (clazz == Integer.TYPE) {
578                 return Integer.class;
579             }
580             if (clazz == Float.TYPE) {
581                 return Float.class;
582             }
583             if (clazz == Double.TYPE) {
584                 return Double.class;
585             }
586             if (clazz == Boolean.TYPE) {
587                 return Boolean.class;
588             }
589             if (clazz == Long.TYPE) {
590                 return Long.class;
591             }
592             if (clazz == Character.TYPE) {
593                 return Character.class;
594             }
595             if (clazz == Short.TYPE) {
596                 return Short.class;
597             }
598             if (clazz == Byte.TYPE) {
599                 return Byte.class;
600             }
601             throw new YAMLException("Unexpected primitive " + clazz);
602         }

604         @SuppressWarnings("unchecked")
605         public void construct2ndStep(Node node, Object object) {
606             SequenceNode snode = (SequenceNode) node;
607             if (List.class.isAssignableFrom(node.getType())) {
608                 List<Object> list = (List<Object>) object;
609                 constructSequenceStep2(snode, list);
610             } else if (node.getType().isArray()) {
611                 constructArrayStep2(snode, object);
612             } else {
613                 throw new YAMLException("Immutable objects cannot be recursive.");
614             }

618     protected Class<?> getClassForNode(Node node) {
619         Class<? extends Object> classForTag = typeTags.get(node.getTag());
620         if (classForTag == null) {
621             String name = node.getTag().getClassName();
622             Class<?> cl;
623             try {
624                 cl = getClassForName(name);
625             } catch (ClassNotFoundException e) {
626                 throw new YAMLException("Class not found: " + name);
627             }
628             typeTags.put(node.getTag(), cl);
629             return cl;
630         } else {
631             return classForTag;
632         }
633     }
